🛠️ Debug Script (Paste in Browser Console)

// Enhanced debug script for hover autoplay and unmute functionality
console.log('🎵 Starting Enhanced Hover Autoplay Debug');

setTimeout(() => {
  console.log('📍 Current URL:', window.location.href);
  
  // Check for video cards
  const videoCards = document.querySelectorAll('[class*="group cursor-pointer"]');
  console.log('📹 Video cards found:', videoCards.length);
  
  if (videoCards.length > 0) {
    const firstCard = videoCards[0];
    
    // Check structure
    const thumbnail = firstCard.querySelector('img[alt]');
    const watchLink = firstCard.querySelector('a[href*="/watch/"]');
    
    console.log('🖼️ Thumbnail:', !!thumbnail);
    console.log('🔗 Watch link:', !!watchLink);
    
    if (watchLink) {
      const videoId = watchLink.href.match(/\/watch\/(.+)$/)?.[1];
      console.log('🆔 Raw video ID:', videoId);
      
      // Test ID extraction logic
      let extractedId = null;
      if (videoId?.startsWith('youtube-')) {
        extractedId = videoId.replace('youtube-', '');
      } else if (videoId?.startsWith('google-search-')) {
        extractedId = videoId.replace('google-search-', '');
      } else if (videoId?.length === 11 && /^[a-zA-Z0-9_-]+$/.test(videoId)) {
        extractedId = videoId;
      }
      
      console.log('🎬 Extracted YouTube ID:', extractedId);
      console.log('✅ Valid for autoplay:', extractedId && extractedId.length === 11);
      
      if (extractedId && extractedId.length === 11) {
        console.log('🎯 Testing hover autoplay and unmute...');
        
        // Simulate hover
        firstCard.dispatchEvent(new MouseEvent('mouseenter', { bubbles: true }));
        
        setTimeout(() => {
          const badge = firstCard.querySelector('[class*="bg-red-600"]');
          console.log('🏷️ Preview badge:', !!badge);
          
          setTimeout(() => {
            const iframe = firstCard.querySelector('iframe[src*="youtube-nocookie.com"]');
            const muteButton = firstCard.querySelector('button[title*="mute"], button[title*="Mute"]');
            
            console.log('🎬 Iframe loaded:', !!iframe);
            console.log('🔊 Mute button found:', !!muteButton);
            
            if (iframe) {
              console.log('✅ SUCCESS! Iframe src:', iframe.src);
              console.log('🔇 Video starts muted:', iframe.src.includes('mute=1'));
              
              if (muteButton) {
                console.log('🎵 Mute button available - testing click...');
                console.log('🔊 Button title:', muteButton.getAttribute('title'));
                
                // Test button click
                setTimeout(() => {
                  console.log('🖱️ Clicking unmute button...');
                  muteButton.click();
                  
                  setTimeout(() => {
                    const newIframe = firstCard.querySelector('iframe[src*="youtube-nocookie.com"]');
                    if (newIframe) {
                      console.log('🎵 After unmute - iframe src:', newIframe.src);
                      console.log('🔊 Video should be unmuted:', newIframe.src.includes('mute=0'));
                    }
                  }, 1000);
                }, 1000);
              } else {
                console.log('❌ No mute button found');
              }
            } else {
              console.log('❌ No iframe found');
            }
            
            // Clean up after test
            setTimeout(() => {
              firstCard.dispatchEvent(new MouseEvent('mouseleave', { bubbles: true }));
              console.log('✅ Test completed - cleaned up hover state');
            }, 5000);
          }, 700);
        }, 200);
      } else {
        console.log('❌ Invalid video ID - hover won\'t work');
      }
    }
  } else {
    console.log('❌ No video cards found');
    console.log('💡 Make sure you\'re on the home page or search page with videos loaded');
  }
}, 2000);

console.log('✅ Enhanced debug script loaded - waiting 2 seconds...');

🔧 Troubleshooting

Common Issues:

  • • No video cards: Search results not loaded
  • • No autoplay: Invalid video IDs
  • • No unmute button: Iframe not loading
  • • Button not working: JavaScript errors

Solutions:

  • • Wait longer for page load
  • • Check browser console for errors
  • • Verify API keys in .env.local
  • • Try different videos/searches
